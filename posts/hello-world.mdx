func FizzBuzz(n int) {

		s := ""
		switch [2]bool{i%3 == 0, i%5 == 0} {
		case [2]bool{true, true}:

		case [2]bool{false, true}:
			s = "Buzz"

		}
		fmt.Println(s)
	}
}
